--创建xjobs_task表
CREATE TABLE xjobs_task (
    job_id       INTEGER     PRIMARY KEY ASC AUTOINCREMENT
                             NOT NULL,
    job_name     TEXT,
    command_lang TEXT,
    command      TEXT,
    input_param  TEXT,
    comments     TEXT,
    cron_exp     TEXT,
    is_pause     INTEGER (1) DEFAULT (0),
    success_exit TEXT,
    update_time  DATETIME,
    job_type     TEXT,
    job_subtype  TEXT
);

--创建xjobs_dictionary表
CREATE TABLE xjobs_dictionary (
    l_dictionary_no      INTEGER     NOT NULL,
    vc_item_index        TEXT        NOT NULL,
    vc_item_name         TEXT,
    l_order              INTEGER (8),
    parent_dictionary_no INTEGER,
    parent_item_index    TEXT,
    vc_bz                TEXT,
    PRIMARY KEY (
        l_dictionary_no,
        vc_item_index
    )
);
